import socket 
import re
import codecs

hote = "challenge01.root-me.org"
port = 52021

client = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
client.connect((hote, port))

donnees = client.recv(2048)
message = donnees.decode('utf-8')
print (message)

resultat = re.findall(r"'(.*?)'", message)
txt = resultat[0]

resultat = codecs.decode(txt, 'rot_13')
print(resultat)

client.send((resultat + '\n').encode('utf-8'))
reponse = client.recv(2048)
print(reponse.decode('utf-8'))

import socket 
import math
import re
import struct

hote = "challenge01.root-me.org"
port = 52002

client = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
client.connect((hote, port))

donnees = client.recv(2048)
message = donnees.decode('utf-8')
print (message)
chiffres = re.findall(r'\d+', message)
print (chiffres)

if len(chiffres) >= 2:
    nombre1 = int(chiffres[-2])
    nombre2 = int(chiffres[-1])

r_c = math.sqrt(nombre1)
nombre3 = r_c*nombre2
resultat = format(nombre3,'.2f')


client.send((resultat + '\n').encode('utf-8'))
reponse = client.recv(2048)
print(reponse.decode('utf-8'))
